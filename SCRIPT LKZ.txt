local library = loadstring(game:HttpGet('https://raw.githubusercontent.com/cypherdh/VanisUILIB/main/.gitignore'))()

local Window = library:CreateWindow("Roblox ESP", "v1.0", 10044538000)

local Tab = Window:CreateTab("Configurações")

local Page = Tab:CreateFrame("Opções")

local highlighting = false
local highlightColor = Color3.new(1, 0, 0) -- Cor padrão para destaque (vermelho)

-- Função para destacar todos os jogadores com uma cor específica
local function HighlightAllPlayers(color)
    for _, player in pairs(game.Players:GetPlayers()) do
        local character = player.Character
        if character then
            local humanoid = character:FindFirstChildOfClass("Humanoid")
            if humanoid then
                -- Criar um efeito visual de destaque (exemplo: uma luz)
                local highlight = Instance.new("PointLight")
                highlight.Color = color
                highlight.Brightness = 5
                highlight.Range = 15
                highlight.Parent = character
                -- Pausar o efeito após um tempo
                wait(2)
                highlight:Destroy()
            end
        end
    end
end

-- Botão para destacar todos os jogadores
local HighlightButton = Page:CreateButton("Highlight All Players", "Destaca todos os jogadores no jogo", function()
    if not highlighting then
        highlighting = true
        HighlightAllPlayers(highlightColor)
    else
        highlighting = false
        -- Remover todos os efeitos de destaque se desativar
        for _, player in pairs(game.Players:GetPlayers()) do
            local character = player.Character
            if character then
                for _, obj in ipairs(character:GetChildren()) do
                    if obj:IsA("PointLight") then
                        obj:Destroy()
                    end
                end
            end
        end
    end
end)

-- Seletor de cor para o destaque
local ColorPicker = Page:CreateColorPicker("Highlight Color", highlightColor, function(color)
    highlightColor = color
    -- Atualizar a cor para o destaque se estiver ativado
    if highlighting then
        HighlightAllPlayers(highlightColor)
    end
end)
