local library = loadstring(game:HttpGet("https://raw.githubusercontent.com/drillygzzly/Other/main/1"))()

library:init() -- Inicializa a biblioteca, não delete isso

local Window = library.NewWindow({
	title = "Informant.Wtf",
	size = UDim2.new(0, 525, 0, 650)
})

local tabs = {
    Tab1 = Window:AddTab("Tab1"),
	Settings = library:CreateSettingsTab(Window),
}

-- Adicionando uma seção para o destaque de jogadores
local sections = {
	Section1 = tabs.Tab1:AddSection("ESP Settings", 1),
	HighlightSection = tabs.Tab1:AddSection("Highlight Players", 2), -- Nova seção para o destaque
}

local highlighting = false
local highlightColor = Color3.new(1, 0, 0) -- Cor padrão para destaque (vermelho)

-- Função para destacar todos os jogadores com uma cor específica
local function HighlightAllPlayers(color)
    for _, player in pairs(game.Players:GetPlayers()) do
        local character = player.Character
        if character then
            local humanoid = character:FindFirstChildOfClass("Humanoid")
            if humanoid then
                -- Criar um efeito visual de destaque (exemplo: uma luz)
                local highlight = Instance.new("PointLight")
                highlight.Color = color
                highlight.Brightness = 5
                highlight.Range = 15
                highlight.Parent = character
                -- Pausar o efeito após um tempo
                wait(2)
                highlight:Destroy()
            end
        end
    end
end

-- Botão para iniciar/parar o destaque de jogadores
local HighlightButton = sections.HighlightSection:AddButton({
	enabled = true,
	text = "Highlight Players",
	flag = "HighlightPlayers",
	tooltip = "Highlights all players in the game",
	risky = false,
	confirm = false,
	callback = function()
	    if not highlighting then
	        highlighting = true
	        HighlightAllPlayers(highlightColor)
	    else
	        highlighting = false
	        -- Remover todos os efeitos de destaque se desativar
	        for _, player in pairs(game.Players:GetPlayers()) do
	            local character = player.Character
	            if character then
	                for _, obj in ipairs(character:GetChildren()) do
	                    if obj:IsA("PointLight") then
	                        obj:Destroy()
	                    end
	                end
	            end
	        end
	    end
	end
})

-- Seletor de cor para o destaque
local ColorPicker = sections.HighlightSection:AddColor({
    enabled = true,
    text = "Highlight Color",
    flag = "HighlightColor",
    tooltip = "Select highlight color",
    color = highlightColor,
    trans = 0,
    open = false,
    callback = function(color)
        highlightColor = color
        -- Atualizar a cor para o destaque se estiver ativado
        if highlighting then
            HighlightAllPlayers(highlightColor)
        end
    end
})

library:SendNotification("Notification", 5, Color3.new(255, 0, 0)) -- Exemplo de notificação (opcional)
