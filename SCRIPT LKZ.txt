local OrionLib = loadstring(game:HttpGet(('https://raw.githubusercontent.com/shlexware/Orion/main/source')))()

OrionLib:MakeNotification({
    Name = "Orion Example",
    Content = "Orion Example",
    Image = "rbxassetid://4483345998",
    Time = 5
})

local Window = OrionLib:MakeWindow({Name = "Orion Example", HidePremium = false, SaveConfig = true, ConfigFolder = "Orion"})

-- Player Tab --
local PlayerTab = Window:MakeTab({
    Name = "Player",
    Icon = "rbxassetid://4483345998",
    PremiumOnly = false
})

local PlayerSection = PlayerTab:AddSection({
    Name = "Player"
})

PlayerSection:AddSlider({
    Name = "Walkspeed",
    Min = 16,
    Max = 100,
    Default = 5,
    Color = Color3.fromRGB(255, 255, 255),
    Increment = 1,
    ValueName = "Walkspeed",
    Callback = function(Value)
        game.Players.LocalPlayer.Character.Humanoid.WalkSpeed = Value
    end
})
-- Player Tab End --

-- Settings Tab --
local SettingsTab = Window:MakeTab({
    Name = "Settings",
    Icon = "rbxassetid://4483345998",
    PremiumOnly = false
})

local SettingsSection = SettingsTab:AddSection({
    Name = "Settings"
})

SettingsSection:AddButton({
    Name = "Destroy UI",
    Callback = function()
        OrionLib:Destroy()
    end
})
-- Settings Tab End --

-- Highlight Players Tab --
local HighlightTab = Window:MakeTab({
    Name = "Highlight",
    Icon = "rbxassetid://4483345998",
    PremiumOnly = false
})

local HighlightSection = HighlightTab:AddSection({
    Name = "Highlight All Players"
})

local highlightedPlayers = {}

-- Função para criar e destacar um jogador com opções de personalização
local function highlightPlayer(player, color, thickness, filled)
    if player ~= game.Players.LocalPlayer then
        local character = player.Character
        if character then
            local humanoid = character:FindFirstChildOfClass("Humanoid")
            if humanoid then
                -- Criar um SelectionBox personalizado
                local highlightBox = Instance.new("SelectionBox")
                highlightBox.Adornee = character:WaitForChild("HumanoidRootPart")

                -- Opções de personalização
                highlightBox.Color3 = color or Color3.new(0, 1, 0) -- Cor padrão: verde
                highlightBox.LineThickness = thickness or 0.05 -- Espessura padrão da linha
                highlightBox.SurfaceTransparency = filled and 0 or 1 -- Preenchimento opcional

                highlightBox.Parent = character
                table.insert(highlightedPlayers, highlightBox)
            end
        end
    end
end

HighlightSection:AddColorPicker({
    Name = "Highlight Color",
    Default = Color3.fromRGB(0, 255, 0), -- Cor verde padrão
    Callback = function(color)
        highlightColor = color
    end
})

HighlightSection:AddSlider({
    Name = "Line Thickness",
    Min = 0.01,
    Max = 0.1,
    Default = 0.05,
    ValueName = "Thickness",
    Callback = function(value)
        highlightThickness = value
    end
})

HighlightSection:AddToggle({
    Name = "Fill Box",
    Default = false,
    Callback = function(enabled)
        highlightFilled = enabled
    end
})

HighlightSection:AddButton({
    Name = "Highlight All",
    Callback = function()
        for _, player in ipairs(game.Players:GetPlayers()) do
            highlightPlayer(player, highlightColor, highlightThickness, highlightFilled)
        end
    end
})

-- Função para limpar todos os destaques
local function clearHighlights()
    for _, box in ipairs(highlightedPlayers) do
        box:Destroy()
    end
    highlightedPlayers = {}
end

HighlightSection:AddButton({
    Name = "Clear Highlight",
    Callback = function()
        clearHighlights()
    end
})
-- Highlight Players Tab End --

OrionLib:Init() -- Finaliza a inicialização da UI Lib
