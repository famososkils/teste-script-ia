-- Script para destacar todos os jogadores em um jogo do Roblox
local Players = game:GetService("Players")
local Highlight = Instance.new("Highlight")
Highlight.FillColor = Color3.fromRGB(0, 0, 255) -- Cor amarela
Highlight.FillTransparency = 0.5 -- Semitransparente
Highlight.OutlineColor = Color3.fromRGB(0, 0, 0) -- Contorno branco

-- Loop através de todos os jogadores
for _, player in pairs(Players:GetPlayers()) do
    local character = player.Character
    if character then
        -- Adicione o destaque ao personagem do jogador
        Highlight:Clone().Parent = character
    end
end

-- Crie uma janela com uma aba e um botão
local OrionLib = loadstring(game:HttpGet(('https://raw.githubusercontent.com/shlexware/Orion/main/source')))()

local Window = OrionLib:MakeWindow({Name = "Orion Example", HidePremium = false, SaveConfig = true, ConfigFolder = "Orion"})

-- Aba de Jogador
local PlayerTab = Window:MakeTab({
    Name = "Player",
    Icon = "rbxassetid://4483345998",
    PremiumOnly = false
})

local PlayerSection = PlayerTab:AddSection({
    Name = "Player"
})

PlayerSection:AddSlider({
    Name = "Walkspeed",
    Min = 16,
    Max = 100,
    Default = 5,
    Color = Color3.fromRGB(255, 255, 255),
    Increment = 1,
    ValueName = "Walkspeed",
    Callback = function(Value)
        game.Players.LocalPlayer.Character.Humanoid.WalkSpeed = Value
    end
})
